 22 /**
 23  * The Wizard class handles all of the mechanics behind how the wizard 
 24  * works.
 25  * @constructor
 26  */
 27 var Wizard = function() {
 28     var dialog = new goog.ui.Dialog();
 29     dialog.setButtonSet(false);
 30     dialog.setHasTitleCloseButton(false);
 31     dialog.setEscapeToCancel(false);
 32     dialog.setTitle("Please wait...");
 33     dialog.setContent(''+
 34             '<div class="loading-message">'+
 35               '<p>Currently fetching content for this page...</p>'+
 36               '<img src="/css/custom-theme/images/loading_indeterminate.gif" '+
 37                    'width="500" height="22" />'+
 38             '</div>');
 39     /**
 40      * The loading dialog.
 41      * @type {goog.ui.Dialog}
 42      */
 43     this.loading_dialog_ = dialog;
 44     
 45     this.current_page = $$('div', 'current-page')[0];
 46     this.setupPageStyle(this.current_page);
 47     //this.configLinks(this.current_page);
 48     try {
 49         this.history = new goog.history.Html5History();
 50         this.history.setUseFragment(false);
 51         this.history.setPathPrefix("/wizard/page/");
 52     } catch (e) {
 53         this.history = new goog.History();
 54     }
 55     this.history.setEnabled(true);
 56     var $this = this;
 57     goog.events.listen(this.history,goog.history.EventType.NAVIGATE,function(e){
 58         $this.processHistoryEvent(e);
 59     });
 60     this.registerButtons();
 61 }

 63 /**
 64  * Processes all history events from the goog.history.EventType.NAVIGATE event.
 65  * @this {Wizard}
 66  * @param {Event} evt The event to handle.
 67  */
 68 Wizard.prototype.processHistoryEvent = function(evt) {
 69     var tok = evt.token;
 70     var prevTok = parseInt(goog.dom.getTextContent($('page-number')));
 71     if (!tok) {
 72         this.history.replaceToken(1);
 73         tok = "1";
 74     } else if (!goog.string.isNumeric(tok)) {
 75         throw Error("History tokens must be numeric.");
 76     }
 77     tok = parseInt(tok);
 78     if (tok == prevTok) return;
 79     goog.dom.setTextContent($('page-number'), tok);
 80     var href = '/ajax/wizard/page/'+tok;
 81     var $this = this;
 82     goog.net.XhrIo.send(href, function(netEvt) {
 83         var xhr = netEvt.target;
 84         var new_page = goog.dom.createDom('div', 'page');
 85         var wrapper = goog.dom.createDom('div', 'page-wrapper');
 86         goog.dom.appendChild(new_page, wrapper);
 87         var inner = goog.dom.createDom('div', 'page-content');
 88         goog.dom.appendChild(wrapper, inner);
 89         inner.innerHTML = xhr.getResponseText();
 90 
 91         $this.updateWizard(new_page, tok > prevTok);
 92     });
 93 }

 95 /**
 96  * Updates the wizard to the given page.
 97  * @this {Wizard}
 98  * @param {Element} dom The page to update.
 99  * @param {fromRight} Whether or not to bring the page in from the right.
100  */
101 Wizard.prototype.updateWizard = function(dom, fromRight) {
102     var parent = $$(null, 'content')[0];
103     dom.style.position = "absolute";
104     dom.style.display = "block";
105     var pos_r = goog.style.getPosition($$(null, 'right-page')[0]);
106     var pos_l = Wizard.LEFT_POSITION;
107     var pos_c = goog.style.getPosition(this.current_page);
108     var flyIn = new goog.fx.AnimationParallelQueue();
109     var coord_r = [pos_r.x, pos_r.y];
110     var coord_c = [pos_c.x, pos_c.y];
111     var coord_l = [pos_l.x, pos_l.y];
112     var old_end, new_start;
113     if (fromRight) {
114         goog.style.setPosition(dom, pos_r);
115         old_end = coord_l;
116         new_start = coord_r;
117     } else {
118         goog.style.setPosition(dom, pos_l);
119         old_end = coord_r;
120         new_start = coord_l;
121     }
122     var $this = this;
123     goog.dom.appendChild(parent, dom);
124     flyIn.add(new goog.fx.dom.Slide(this.current_page, coord_c, old_end, 
125                 Wizard.ANIMATION_LENGTH, Wizard.ANIMATION_EASING));
126     flyIn.add(new goog.fx.dom.Slide(dom, new_start, coord_c,
127                 Wizard.ANIMATION_LENGTH, Wizard.ANIMATION_EASING));
128     goog.events.listen(flyIn, goog.fx.Transition.EventType.BEGIN, function(e) {
129         if (fromRight) {
130             $this.registerNextButton(dom);
131         } else {
132             $this.registerPrevButton(dom);
133         }
134     });
135     goog.events.listen(flyIn, goog.fx.Transition.EventType.END, function(e) {
136         goog.dom.removeNode($this.current_page);
137         goog.dom.classes.add(dom, 'current-page');
138         if (fromRight) {
139             $this.registerPrevButton(dom);
140         } else {
141             $this.registerNextButton(dom);
142         }
143         $this.registerFormSubmit(dom);
144         $this.current_page = dom;
145         $this.setLoading(false);
146     });
147     flyIn.play();
148 };

159 /**
160  * Registers the form to submit (if any)
161  * @this {Wizard}
162  * @param {Element} opt_page The page to register.  If none, current_page is
163  *      used.
164  */
165 Wizard.prototype.registerFormSubmit = function(opt_page) {
166     var page = opt_page || this.current_page;
167     var form = $$('form', null, page);
168     var $this = this;
169     if (form && form.length > 0) {
170         form = form[0];
171         goog.events.listen(form, goog.events.EventType.SUBMIT, function(evt) {
172             $this.setLoading(true);
173             evt.preventDefault();
174             var in_data = goog.dom.forms.getFormDataString(form);
175             var callback = function(evt) {
176                 var xhr = evt.target;
177                 var out_data = xhr.getResponseJson();
178                 console.log(xhr.getResponseText());
179                 console.log(xhr.getResponseType());
180                 console.log(out_data);
181                 if (out_data.success) {
182                     // Move on to the next thing.
183                     var current_page = parseInt(goog.dom.getTextContent(
184                             $('page-number')));
185                     $this.updateHistory(current_page+1);
186                 } else {
187                     // Display error message.
188                     var message = out_data.reason;
189                     var flash = $('flash');
190                     flash.style.color = "red";
191                     flash.style.fontWeight = "bold";
192                     flash.style.textAlign = "center";
193                     goog.dom.setTextContent(flash, message);
194                 }
195             };
196             goog.net.XhrIo.send("/ajaxwizform.php", callback, "POST", in_data);
197         });
198     } 
199 };

201 /**
202  * Registers the previous button on the page.
203  * @this {Wizard}
204  * @param {Element=} opt_page Option page.  this.current_page used otherwise.
205  */
206 Wizard.prototype.registerPrevButton = function(opt_page) {
207     var page = opt_page || this.current_page;
208     var prev_link = $$(null, 'prev-link', page);
209     var $this = this;
210     if (prev_link && prev_link.length > 0 && prev_link[0].rel != 'last') {
211         prev_link = prev_link[0];
212         $$('div', 'left-page')[0].style.visibility = 'visible';
213     } else {
214         prev_link = null;
215         $$('div', 'left-page')[0].style.visibility = 'hidden';
216     }
217     if (prev_link) {
218         goog.events.listen(prev_link, goog.events.EventType.CLICK, function(e) {
219             if (prev_link.nodeName.toUpperCase()=="A") {
220                 e.preventDefault();
221                 var href = new goog.Uri(prev_link.href).getPath();
222                 if (goog.string.startsWith(href, '/wizard/page/')) {
223                     href = href.substring(13);
224                 }
225                 if (goog.string.endsWith(href, '/')) {
226                     href = href.substring(0, href.length-1);
227                 }
228                 $this.updateHistory(href);
229             } 
230         });
231     }
232 };
