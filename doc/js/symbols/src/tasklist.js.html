103 /**
104  * Static function for return the path of an href.  If the passed element
105  * is a string, the value is just returned.
106  * @static
107  * @param {(Element|string)} anchor An anchor to grab the path from.
108  * @return {string} The path part of a URI.
109  */
110 Util.anchorToHref = function(anchor) {
111     if (goog.isString(anchor)) {
112         return anchor;
113     } else {
114         return new goog.Uri(anchor.href).getPath();
115     }

119 /**
120  * Static function for trimming a string with a given set of characters at the 
121  * beginning and end of a string.
122  * @static
123  * @param {string} string The string to trim
124  * @param {string=} opt_chars An optional string containing characters to trim.
125  * @return {string} The trimmed string.
126  */
127 Util.trim = function(string, opt_chars) {
128     var specials = new RegExp("[.*+?|()\\[\\]{}\\\\]", "g");
129     var chars = opt_chars ||"["+opt_chars.replace(specials, "$&")+"]";
130     var re = new RegExp("^"+chars+"*(.*?)"+chars+"*$","g");
131     return string.replace(re, "$1");
132 };

177 /**
178  * Handles the NAVIGATE event for the history object.
179  * @this {Util}
180  * @param {Event} evt The event to handle.
181  */
182 Util.prototype.historyHandler = function(evt) {
183     this.setLoading(true);
184     if (evt.token) {
185         var token = Util.trim(evt.token, '/').toLowerCase();
186         var $this = this;
187         var eventType;
188 
189         if (goog.string.endsWith(token, "edit")) {
190             eventType = Util.EventType.EDIT;
191             this.getTaskEditor(evt.token);
192         } else if (goog.string.endsWith(token, "copy")) {
193             eventType = Util.EventType.COPY;
194             this.getTaskEditor(evt.token);
195         } else if (goog.string.endsWith(token, "new")) {
196             eventType = Util.EventType.NEW;
197             this.getTaskEditor(evt.token);
198         } else if (goog.string.endsWith(token, "delete")) {
199             eventType = Util.EventType.DELETE;
200         } else if (token.split("/").length-1 == 0) {
201             eventType = Util.EventType.TASKLIST;
202             this.getTaskList(evt.token);
203         } else {
204             eventType = Util.EventType.VIEW;
205             this.getTaskDescription(evt.token);
206         }
207     } else {
208         // Looks like the token is null (at the root view), so lets update it
209         var params = [['li', 'selected', 'tasks-list'],
210                       ['li', null, 'tasks-list'],
211                       ['li', 'selected', 'accounts-list'],
212                       ['li', null, 'accounts-list']];
213         for (var i = 0; i < params.length; i++) {
214             var p = params[i];
215             var li = goog.dom.getElementsByTagNameAndClass(p[0], p[1], $(p[2]));
216             if (li.length != 0) {
217                 var a = goog.dom.getElementsByTagNameAndClass('a', null, li[0]);
218                 this.setSelected(li[0]);
219                 this.replaceHistory(Util.anchorToHref(a[0]));
220                 return;
221             }
222         }
223         // We should never ever ever make it here.
224         throw "wtf";
225     }
226 };

272 Util.registerDatePicker = function(widget, input_wrapper, opt_format) {
273     var dp = new goog.ui.DatePicker();
274     var input = goog.dom.getElementsByTagNameAndClass('input', null, 
275             input_wrapper)[0];
276     var image = goog.dom.getElementsByTagNameAndClass(null, 'date-icon',

279     /**
280      * Function for hiding the datepicker.
281      * @type {function()}
282      */
283     var hideDatePicker = function() {
284         goog.style.showElement(widget, false);

287     /**
288      * Function for showing the datepicker

290      */
291     var showDatePicker = function() {
292         goog.style.showElement(widget, true);

294

298     /**
299      * Functoin for setting the date on the datepicker to the value from the

301      * @type {function()}
302      */
303     var setDate = function() {
304         var d = new goog.date.Date();
305         var value = goog.string.trim(input.value);
306         var parsed = parser.strictParse(value, d);
307         input.value = value;
308         if (value.length > 0) {

311     }

314 
315     if (!widget.id) {
316         widget.id = "goog-datepicker-"+goog.getUid(widget);

321     dp.setAllowNone(false);
322     dp.setUseSimpleNavigationMenu(true);

326     });
327

330     goog.events.listen(document,goog.events.EventType.MOUSEDOWN, function(evt) {

341         hideDatePicker();
342     });
343 
344     goog.events.listen(input, goog.events.EventType.KEYDOWN, function(evt) {
345         if (evt.keyCode == goog.events.KeyCodes.TAB) {
346             hideDatePicker();
347         }
348     });

655         evt.preventDefault();
656         var form = $$('form', null, 
657                 $this.account_dialog_.getContentElement())[0];
658         var in_data = goog.dom.forms.getFormDataString(form);
659         $this.account_dialog_.setVisible(false);
660         if (!evt.key || evt.key != 'ok') {
661             goog.array.forEach($$('input', null, form), function(input) {
662                 input.value = '';
663             });
664             return;
665         }
666         $this.setLoading(true);
667         var callback = function(evt) {
668             var xhr = evt.target;
669             var out_data = xhr.getResponseJson();
670             if (out_data['success']) {
671                 // Move on to the next thing.
672                 $this.addAccount(out_data['account']);
673                 goog.array.forEach($$('input', null, form), function(input) {
674                     input.value = '';
675                 });
676             } else {
677                 $this.account_dialog_.setVisible(true);
678                 // Display error message.
679                 var message = out_data['reason'];
680                 var flash = $$('div', 'flash', form)[0];
681                 flash.style.color = "red";
682                 flash.style.fontWeight = "bold";
683                 flash.style.textAlign = "center";
684                 goog.dom.setTextContent(flash, message);
685             }
686             $this.setLoading(false);
687         };
688         goog.net.XhrIo.send("/ajaxwizform.php", callback, "POST", in_data);
689     };
690 
691     goog.events.listen(form, goog.events.EventType.SUBMIT, formsubmit);
692     goog.events.listen(this.account_dialog_, goog.ui.Dialog.EventType.SELECT,
693             formsubmit);

766 /**
767  * Registers the message with flash viewer (banner across top).
768  * @this {Util}
769  * @param {string=} opt_message Optional message to use.  Otherwise, just sets 
770  *              events.
771  * @param {int=} opt_speed Optional speed to use.
772  */
773 Util.prototype.registerFlashMessage = function(opt_message, opt_speed) {
774     var flash = $('flash');
775     var flash_inner = $('flash-inner');
776     var size, padbox;
777     flash.style.overflow = "hidden";
778     flash_inner.style.overflow = "hidden";
779     var speed = opt_speed || 300;
780     if (opt_message) {
781         flash_inner.innerHTML = opt_message;
782         
783         if (!goog.style.isElementShown(flash)) {
784             flash.style.height = "auto";
785             flash_inner.style.height = "auto";
786             size = goog.style.getSize(flash);
787             padbox = goog.style.getPaddingBox(flash);
788             size.height -= padbox.top + padbox.bottom;
789             size.width -= padbox.left + padbox.right;
790             goog.style.setSize(flash, size.width, 0);
791             goog.style.showElement(flash, true);
792 
793             //Animate the flash message by sliding downwards.
794             var show = new goog.fx.dom.ResizeHeight(flash, 0, size.height, 
795                     speed);
796             show.play();
797         }
798     }
799     
800     size = goog.style.getSize(flash);
801     padbox = goog.style.getPaddingBox(flash);
802     size.height -= padbox.top + padbox.bottom;
803     size.width -= padbox.left + padbox.right;
804     var timeout;
805     var slideUpAndHide = function () {
806         var hide = new goog.fx.dom.ResizeHeight(flash, size.height, 0, speed);
807         goog.events.listen(hide, goog.fx.Animation.EventType.END, function() {
808             flash.style.display = "none";
809             flash.style.height = "auto";
810         });
811         hide.play();
812     };
813     var listener = function() {
814         clearTimeout(timeout);
815         slideUpAndHide();
816     };
817 
818     // Animate the flash message by sliding upwards.
819     var timeout = setTimeout(function() {
820         goog.events.unlisten(flash, goog.events.EventType.CLICK, listener);
821         slideUpAndHide();
822     }, Util.FLASH_MESSAGE_DISPLAY_SECONDS*1000);
823     goog.events.listenOnce(flash, goog.events.EventType.CLICK, listener);
824 }

826 /**
827  * Registers a menu as collapsible.
828  * @param {Element} menu to set as collapsible
829  * @param {int=} opt_speed The speed to set. Optional.
830  */
831 Util.registerCollapsibleMenu = function(menu, opt_speed) {
832     var header = goog.dom.getElementsByTagNameAndClass('span', 'header', 
833             menu)[0];
834     var list   = goog.dom.getElementsByTagNameAndClass('ul', 'list', menu)[0];
835     var collapser = goog.dom.createDom('span', 'ui-icon ui-icon-triangle-1-s');
836     var li = goog.dom.getElementsByTagNameAndClass('li', null, list);
837     var speed = opt_speed || 300;
838     goog.dom.insertChildAt(header, collapser, 0);
839 
840     goog.events.listen(header, goog.events.EventType.CLICK, function(evt) {
841         var anim, height;
842         if (goog.dom.classes.has(list, 'collapsed-list')) {
843             // UNCOLLAPSE
844             list.style.overflow = "";
845             height = goog.style.getSize(list).height;
846             list.style.display = "";
847             anim = new goog.fx.dom.ResizeHeight(list, 0, height, speed);
848             goog.dom.classes.addRemove(collapser, 'ui-icon-triangle-1-e',
849                 'ui-icon-triangle-1-s');
850             goog.events.listen(anim, goog.fx.Animation.EventType.END, 
851                     function() {
852                 goog.dom.classes.remove(list, 'collapsed-list');
853                 list.style.height = "";
854             });
855         } else {
856             // COLLAPSE
857             list.style.overflow = "hidden";
858             height = goog.style.getSize(list).height;
859             anim = new goog.fx.dom.ResizeHeight(list, height, 0, speed);
860             goog.dom.classes.addRemove(collapser, 'ui-icon-triangle-1-s',
861                 'ui-icon-triangle-1-e');
862             goog.events.listen(anim, goog.fx.Animation.EventType.END, 
863                     function() {
864                 goog.dom.classes.add(list, 'collapsed-list');
865                 list.style.display = "none";
866                 list.style.height = "";
867             });
868         }
869         anim.play();
870     });
871 };
